8class VAE(nn.Module):
9 '''
10 Variational Autoencoder (VAE) implemented using linear layers.
11 It encodes input data to a latent space representation and then decodes it back to the original input space.
12 The VAE consists of two main parts: an encoder and a decoder. Use dense layers to connect them.
13 '''
14 def __init__(self):
15 super().__init__()
16 d = 20 # Dimension of latent space vector
17 # Encoder network: Maps inputs to the latent space representation.
18 # The encoder outputs two things: mean (mu) and log variance (logvar) of the latent space distribution.
19 self.encoder = nn.Sequential(
20 nn.Linear(784, d ** 2),
21 nn.ReLU(),
22 nn.Linear(d ** 2, d * 2)
23 )
24 # Decoder network: Maps latent space representations back to the input space.
25 self.decoder = nn.Sequential(
26 nn.Linear(d, d ** 2),
27 nn.ReLU(),
28 nn.Linear(d ** 2, 784),
29 nn.Sigmoid(),
30 )
31
32 def reparameterise(self, mu, logvar):
33 '''
34 Reparameterizes the latent space vector to allow backpropagation through random operations.
35 This is done by sampling from the distribution defined by mu and logvar using the reparameterization trick.
36
37 Parameters:
38 - mu: Mean of the latent space distribution.
39 - logvar: Log variance of the latent space distribution.
40
41 Returns:
42 - A sampled latent vector (z).
43 '''
44 if self.training: # Only add randomness during training.
45 std = logvar.mul(0.5).exp_()# Calculates the standard deviation.
46 eps = std.new_empty(std.size()).normal_() # Samples epsilon from a standard normal distribution.
47 return eps.mul_(std).add_(mu) # Returns z = mu + epsilon*std.
48 else:
49 return mu # During evaluation, just use the mean.
50
51 def forward(self, x):
52 '''
53 Defines the forward pass of the VAE.
54
55 Parameters:
56 - x: Input tensor.
57
58 Returns:
59 - Reconstructed input, mean and log variance of the latent space distribution.
60 '''
61 mu_logvar = self.encoder(x.view(-1, 784)).view(-1, 2, d)
62 mu = mu_logvar[:, 0, :]
63 logvar = mu_logvar[:, 1, :]
64 z = self.reparameterise(mu, logvar)
65 return self.decoder(z), mu, logvar

73class CNN_VAE(nn.Module):
74 '''
75 Convolutional Variational Autoencoder (CNN-VAE) for encoding input data to a latent space representation
76 and decoding it back to the original input space. This class uses convolutional layers for the encoder
77 and convolutional transpose layers for the decoder, with dense layers to connect the latent space representation.
78 '''
79
80 def __init__(self, channel_in, latent_dim, hidden_dims=None):
81 super(CNN_VAE, self).__init__()
82 self.latent_dim = latent_dim # latent space vector
83 self.channel_in = channel_in # number of colors
84
85 if hidden_dims is None:
86 hidden_dims = [32, 64, 128, 256]
87
88 # Encoder: Convolutional layers to encode input into a latent representation
89 modules = []
90 for h_dim in hidden_dims:
91 modules.append(
92 nn.Sequential(
93 nn.Conv2d(channel_in, h_dim, kernel_size=7, stride=2, padding=1),
94 nn.BatchNorm2d(h_dim),
95 nn.LeakyReLU())
96 )
97 channel_in = h_dim
98 self.encoder = nn.Sequential(*modules)
99
100 # Transition between encoder and decoder: Dense layers for mu and log variance of latent space
101 self.fc_mu = nn.Linear(hidden_dims[-1] * 4, self.latent_dim)
102 self.fc_logvar = nn.Linear(hidden_dims[-1] * 4, self.latent_dim)
103
104
105 # Decoder Input: Prepares latent dimension for decoding process
106 self.decoder_input = nn.Linear(self.latent_dim, hidden_dims[-1]*4)
107 modules = []
108 hidden_dims=[256,128,64,64,64]
109 for i in range(len(hidden_dims) - 1):
110 modules.append(
111 nn.Sequential(
112 nn.ConvTranspose2d(hidden_dims[i], hidden_dims[i + 1], kernel_size=3, stride=2, padding=1, output_padding=1),
113 nn.BatchNorm2d(hidden_dims[i + 1]),
114 nn.LeakyReLU()
115 )
116 )
117
118 self.decoder = nn.Sequential(*modules)
119
120
121 # Final layer: Convolutional transpose layer to match the original input's shape and channel
122 self.final_layer = nn.Sequential(
123 nn.ConvTranspose2d(hidden_dims[-1], hidden_dims[-1], kernel_size=31, stride=2, padding=1, output_padding=0),
124 nn.BatchNorm2d(hidden_dims[-1]),
125 nn.LeakyReLU(),
126 nn.Conv2d(hidden_dims[-1], out_channels=self.channel_in, kernel_size=2 , stride=1,padding=0),

130 def encode(self, x):
131 ''' Encodes input tensor into a latent space representation (mu and log_var).
132 Args:
133 x: input tensor [B*C*W*L]
134 Return:
135 mu: mean of latent space vector [B*latent_dim]
136 log_var: log variance of latent space vector [B*latent_dim]
137 '''
138 x = self.encoder(x) #[B*C*W*L]
139 x = torch.flatten(x, start_dim=1) #[B*hidden_dims[-1]*4]
140 mu = self.fc_mu(x)
141 log_var = self.fc_logvar(x)
142 return mu, log_var

144 def decode(self,z):
145 ''' Decodes latent space vectors back into reconstructed images.
146 Args:
147 z: latent space vector [B*latent_dim]
148 Return:
149 x_hat: reconstructed images [B*C*W*L]
150 '''
151 result = self.decoder_input(z)
152 result = result.view(-1, 256, 2, 2) # Here, 256 is the last dimension from `hidden_dims`, adjust accordingly.
153 result = self.decoder(result)
154 x_hat = self.final_layer(result)
155 return x_hat

157 def rereparameterize(self, mu,log_var):
158 '''Reparameterization trick to sample from N(mu, var) from N(0,1),
159 for training purpose, add a random noise log_var to the latent space vector,
160 for evaluation purpose, only use the mean of latent space vector
161
162 Args:
163 mu: mean of latent space vector [B*latent_dim]
164 log_var: log variance of latent space vector [B*latent_dim]
165 Return:
166 a latent space tesnor [B*latent_dim]
167
168 '''
169 if self.training: # only for training purpose
170 std = torch.exp(0.5*log_var)
171 eps = torch.randn_like(std)
172 return eps.mul(std).add_(mu)
173 else:
174 return mu

176 def forward(self,inputs):
177 """ connection of encoder and decoder inputs x and output x_hat
178 Args:
179 inputs: input tensor [B*C*W*L]
180 Return:
181 x_hat: reconstructed images [B*C*W*L]
182 mu: mean of latent space vector [B*latent_dim]
183 log_var: log variance of latent space vector [B*latent_dim]
184 """
185 mu, log_var = self.encode(inputs)
186 z = self.rereparameterize(mu, log_var)
187 x_hat = self.decode(z)
188
189 return x_hat,mu,log_var
