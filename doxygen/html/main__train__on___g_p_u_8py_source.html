89# Define the loss function
90def lossfunc(x,x_hat,mu,logvar,beta):
91 """
92 Computes the Variational Autoencoder (VAE) loss function, combining reconstruction loss and KL divergence.
93
94 Args:
95 x (torch.Tensor): Original input images.
96 x_hat (torch.Tensor): Reconstructed images.
97 mu (torch.Tensor): Mean of the latent variables.
98 logvar (torch.Tensor): Log variance of the latent variables.
99 beta (float): Weight for the KL divergence part of the loss.
100
101 Returns:
102 torch.Tensor: The computed loss value.
103 """
104 recons_loss = nn.functional.binary_cross_entropy(x_hat, x, reduction='sum')
105
106 kl_loss = -0.5 * torch.sum(1 + logvar - mu.pow(2) - logvar.exp())
107 # kl_loss = torch.mean(-0.5 * torch.sum(1 + logvar - mu.pow(2) - logvar.exp(), dim = 1), dim = 0)
108 return recons_loss + beta* kl_loss
